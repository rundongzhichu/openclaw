```typescript
// 列出受控的子代理
export function listControlledSubagentRuns(
  controllerSessionKey: string
): SubagentRunRecord[] {
  const filtered: SubagentRunRecord[] = [];
  for (const entry of sortSubagentRuns(listSubagentRunsForController(controllerSessionKey))) {
    const latest = getLatestSubagentRunByChildSessionKey(entry.childSessionKey);
    if (latest?.runId === entry.runId) {
      filtered.push(entry);
    }
  }
  return filtered;
}

// 调整子代理方向 (steer)
export async function steerControlledSubagentRun(params: {
  cfg: OpenClawConfig;
  controller: ResolvedSubagentController;
  entry: SubagentRunRecord;
  message: string;
}): Promise<SteerResult> {
  // 1. 检查是否有活跃的后代代理
  const pendingDescendants = countPendingDescendantRuns(entry.childSessionKey);
  
  // 2. 如果有后代在运行，唤醒已结束的父母
  if (pendingDescendants > 0 && entry.endedAt) {
    markSubagentRunForSteerRestart(entry.runId);
    return { status: "deferred", reason: "waiting-for-descendants" };
  }
  
  // 3. 调用 Gateway 重启子代理
  const result = await callGateway({
    method: "agent",
    params: {
      key: entry.childSessionKey,
      message: params.message,
      mode: "restart",
    },
  });
  
  return { status: "accepted", runId: result.runId };
}
```

- **Session Key 格式**: `agent:{agentId}:main|subagent:{sessionId}`

- **深度限制**: `DEFAULT_SUBAGENT_MAX_SPAWN_DEPTH`
- **Workspace 继承**: 子代理自动继承父代理工作空间
- **完成通知**: 子代理完成后自动 announce 给父代理

```typescript
export async function createPluginRuntime(
  config: OpenClawConfig
): Promise<PluginRuntime> {
  // 1. 扫描已安装的插件
  const installedPlugins = await scanInstalledPlugins();
  
  // 2. 根据配置筛选启用的插件

    channels,

  };
}

export async function runGlobalHook(
  hookName: string,
  ctx: HookContext
): Promise<void> {
  const runner = getGlobalHookRunner();
  if (runner.hasHooks(hookName)) {
    await runner.runHook(hookName, ctx);
  }
}
```

```typescript
async function authenticateConnection(
  params: ConnectParams
): Promise<AuthResult> {
  // 1. Token 认证
  if (params.token) {
    const isValid = await verifyGatewayToken(params.token);
    if (isValid) {
      return { authenticated: true, role: "operator" };
    }
  }
  
  // 2. Password 认证
  if (params.password) {
    const isValid = await verifyGatewayPassword(params.password);
    if (isValid) {
      return { authenticated: true, role: "operator" };
    }
  }
  
  // 3. Tailscale 身份认证
  if (params.device?.id && isTailscaleEnabled()) {
    const identity = await verifyTailscaleIdentity(params.device.id);
    if (identity) {
      return { authenticated: true, role: "node", deviceId: identity.deviceId };
    }
  }
  
  // 4. Control UI 特殊处理
  if (params.client?.mode === "ui" && isTrustedProxy(params.clientIp)) {
    return { authenticated: true, role: "operator", bypassDeviceAuth: true };
  }
  
  return { authenticated: false };
}
```

```typescript
// Scope 层级
const SCOPES_HIERARCHY = {
  "operator.admin": ["operator.read", "operator.write"],
  "operator.read": [],
  "operator.write": [],
  "node.system": ["node.basic"],
  "node.basic": [],
};

function hasScope(grantedScopes: string[], requiredScope: string): boolean {
  if (grantedScopes.includes(requiredScope)) return true;
  
  // 检查父级 scope
  for (const parentScope of SCOPES_HIERARCHY[requiredScope] || []) {
    if (hasScope(grantedScopes, parentScope)) return true;
  }
  
  return false;
}
```

## 🎯 五、核心数据流

### 1. 消息接收流程

```
1. Channel 插件接收消息 (如 Telegram)
   ↓
2. 通道管理器路由到 Gateway
   ↓
3. Gateway 验证发送者 (pairing/open)
   ↓
4. 根据 routing rules 分配到对应 Agent Workspace
   ↓
5. 创建/更新 Session
   ↓
6. 触发 Agent 运行 (Pi Agent RPC)
   ↓
7. 工具执行 (browser/canvas/nodes)
   ↓
8. 流式响应返回 Channel
```

### 2. 子代理 spawning 流程

```
1. 主 Agent 调用 sessions_spawn 工具
   ↓
2. 检查深度限制 (maxSpawnDepth)
   ↓
3. 创建工作空间 (继承或复制)
   ↓
4. 创建子 Session Key
   ↓
5. 注册到 SubagentRegistry
   ↓
6. 在独立 Lane 中执行
   ↓
7. 完成后自动 announce 给父代理
```
